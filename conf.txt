function calculateTotal() {
const chickenQty = parseInt(document.getElementById('chickenQty').value) || 0;
const paneerQty = parseInt(document.getElementById('paneerQty').value) || 0;
const vegQty = parseInt(document.getElementById('vegQty').value) || 0;

const paneerPrice = 80;
const vegPrice = 70;

function confirmOrder() {
const totalCost = document.getElementById('totalCost').value;

alert("Please calculate the total cost before confirming the order.");

const text = document.getElementById("textInput").value;
const charCount = text.length;

function allowDrop(ev) {
ev.preventDefault();
}
function drag(ev) {
ev.dataTransfer.setData("text", ev.target.id);
}
function drop(ev) {
ev.preventDefault();
var data = ev.dataTransfer.getData("text");
var draggedImage = document.getElementById(data);
ev.target.innerHTML = "";
ev.target.appendChild(draggedImage);
}
</script>

function luhnCheck(cardNumber) {
// Luhn Algorithm implementation
let sum = 0;
let alternate = false;
for (let i = cardNumber.length - 1; i >= 0; i--) {
let digit = parseInt(cardNumber.charAt(i));
if (alternate) {
digit *= 2;
if (digit > 9) {
digit = (digit % 10) + 1;
}
}
sum += digit;
alternate = !alternate;
}
return (sum % 10) === 0;
}

function formatCardNumber(cardNumber) {
return cardNumber.replace(/(\d{4})(?=\d)/g, '$1-');
}
